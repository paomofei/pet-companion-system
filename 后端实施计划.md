# 伴读宠物系统 - 后端实施计划（V4 极简版）

> **V4 极简**：去掉 `pending_draws` 计数器。`pets.pending_draw BOOLEAN`：升级置 true，抽奖后置 false。XP 满格升级保留余数。

---

## 1. 技术栈

| 类别 | 选型 |
|------|------|

| 框架 | Nest.js + TypeScript 5 |
| ORM | Prisma (PostgreSQL 15+) |

## 3. Prisma Schema（V4 极简版）

```prisma
generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

// ==================== 用户 ====================
model User {
  id                Int       @id @default(autoincrement())
  nickname          String    @db.VarChar(20)
  energyBalance     Int       @default(0) @map("energy_balance")
  // V4：移除 pendingDraws，抽奖状态移至 pets 表
  onboardingOption  Int       @default(0) @map("onboarding_option")
  currentStreak     Int       @default(0) @map("current_streak")
  maxStreak         Int       @default(0) @map("max_streak")
  totalEnergyEarned Int       @default(0) @map("total_energy_earned")
  totalTasksDone    Int       @default(0) @map("total_tasks_done")
  createdAt         DateTime  @default(now()) @map("created_at")
  updatedAt         DateTime  @updatedAt @map("updated_at")
  deletedAt         DateTime? @map("deleted_at")

  pet          Pet?
  goals        Goal[]
  tasks        Task[]
  energyLogs   EnergyLog[]
  wishes       Wish[]
  interactions InteractionLog[]
  userBadges   UserBadge[]

  @@map("users")
}

// ==================== 宠物 ====================
model Pet {
  id          Int      @id @default(autoincrement())
  userId      Int      @unique @map("user_id")
  name        String   @db.VarChar(20)
  level       Int      @default(1)
  currentXp   Int      @default(0) @map("current_xp")
  maxXp       Int      @default(100) @map("max_xp")
  appearance  String   @default("cat_default")
  // V4 新增：布尔值，true=有一次待抽奖，false=无
  pendingDraw Boolean  @default(false) @map("pending_draw")
  createdAt   DateTime @default(now()) @map("created_at")
  updatedAt   DateTime @updatedAt @map("updated_at")

  user         User             @relation(fields: [userId], references: [id])
  interactions InteractionLog[]

  @@map("pets")
}

// ==================== 目标 ====================
model Goal {
  id        Int       @id @default(autoincrement())
  userId    Int       @map("user_id")
  icon      String    @default("🎯") @db.VarChar(10)
  title     String    @db.VarChar(20)
  deletedAt DateTime? @map("deleted_at")
  createdAt DateTime  @default(now()) @map("created_at")
  updatedAt DateTime  @updatedAt @map("updated_at")

  user  User   @relation(fields: [userId], references: [id])
  tasks Task[]

  @@index([userId, deletedAt])
  @@map("goals")
}

// ==================== 任务 ====================
model Task {
  id            Int       @id @default(autoincrement())
  userId        Int       @map("user_id")
  goalId        Int?      @map("goal_id")
  title         String    @db.VarChar(50)
  targetDate    String    @map("target_date")
  status        Int       @default(0)             // 0=待完成 1=已完成 2=已漏做
  rewardEnergy  Int       @default(10) @map("reward_energy")
  repeatType    Int       @default(0) @map("repeat_type")
  templateId    Int?      @map("template_id")
  isDelayedCopy Boolean   @default(false) @map("is_delayed_copy")
  completedAt   DateTime? @map("completed_at")
  deletedAt     DateTime? @map("deleted_at")
  createdAt     DateTime  @default(now()) @map("created_at")
  updatedAt     DateTime  @updatedAt @map("updated_at")

  user User  @relation(fields: [userId], references: [id])
  goal Goal? @relation(fields: [goalId], references: [id])

  @@index([userId, targetDate, status])
  @@index([goalId, status])
  @@map("tasks")
}

// ==================== 模板 ====================
model TaskTemplate {
  id           Int      @id @default(autoincrement())
  userId       Int      @map("user_id")
  goalId       Int?     @map("goal_id")
  title        String   @db.VarChar(50)
  rewardEnergy Int      @default(10) @map("reward_energy")
  repeatType   Int      @map("repeat_type")
  isActive     Boolean  @default(true) @map("is_active")
  createdAt    DateTime @default(now()) @map("created_at")
  updatedAt    DateTime @updatedAt @map("updated_at")

  @@map("task_templates")
}

// ==================== 能量流水 ====================
model EnergyLog {
  id           Int      @id @default(autoincrement())
  userId       Int      @map("user_id")
  actionType   String   @map("action_type") @db.VarChar(20)
  amount       Int
  balanceAfter Int      @map("balance_after")
  refId        Int?     @map("ref_id")
  createdAt    DateTime @default(now()) @map("created_at")

  user User @relation(fields: [userId], references: [id])

  @@index([userId, createdAt(sort: Desc)])
  @@map("energy_logs")
}

// ==================== 道具 ====================
model Item {
  id         Int    @id @default(autoincrement())
  name       String @db.VarChar(20)
  icon       String @db.VarChar(10)
  costEnergy Int    @map("cost_energy")
  gainXp     Int    @map("gain_xp")
  sortOrder  Int    @default(0) @map("sort_order")

  interactions InteractionLog[]

  @@map("items")
}

// ==================== 互动流水 ====================
model InteractionLog {
  id         Int      @id @default(autoincrement())
  userId     Int      @map("user_id")
  petId      Int      @map("pet_id")
  itemId     Int      @map("item_id")
  energyCost Int      @map("energy_cost")
  xpGained   Int      @map("xp_gained")
  createdAt  DateTime @default(now()) @map("created_at")

  user User @relation(fields: [userId], references: [id])
  pet  Pet  @relation(fields: [petId], references: [id])
  item Item @relation(fields: [itemId], references: [id])

  @@map("interaction_logs")
}

// ==================== 许愿池 ====================
model Wish {
  id        Int       @id @default(autoincrement())
  userId    Int       @map("user_id")
  icon      String    @default("🎁") @db.VarChar(10)
  title     String    @db.VarChar(20)
  weight    Int       @default(10)
  status    Int       @default(0)            // 0=待抽取 1=已实现 2=已作废
  drawnAt   DateTime? @map("drawn_at")
  deletedAt DateTime? @map("deleted_at")
  createdAt DateTime  @default(now()) @map("created_at")
  updatedAt DateTime  @updatedAt @map("updated_at")

  user User @relation(fields: [userId], references: [id])

  @@map("wishes")
}

// ==================== 勋章 ====================
model Badge {
  id          Int    @id @default(autoincrement())
  code        String @unique @db.VarChar(30)
  name        String @db.VarChar(20)
  description String @db.VarChar(100)
  icon        String @db.VarChar(10)
  category    String @db.VarChar(20)
  threshold   Int

  userBadges UserBadge[]

  @@map("badges")
}

model UserBadge {
  id         Int      @id @default(autoincrement())
  userId     Int      @map("user_id")
  badgeId    Int      @map("badge_id")
  unlockedAt DateTime @default(now()) @map("unlocked_at")

  user  User  @relation(fields: [userId], references: [id])
  badge Badge @relation(fields: [badgeId], references: [id])

  @@unique([userId, badgeId])
  @@map("user_badges")
}
```

## 4. 核心事务逻辑

### 4.1 任务打卡

```text
BEGIN TRANSACTION
  1. SELECT task WHERE id=:taskId AND userId=:userId AND status=0
  2. UPDATE tasks SET status=1, completedAt=NOW()
  3. UPDATE users SET energy_balance += :reward (原子)
  4. UPDATE users SET total_tasks_done += 1, total_energy_earned += :reward
  5. 连续打卡：today首打卡 → current_streak++ → 比较 max_streak
  6. INSERT energy_logs
  7. IF goal_id → 重算进度 → 100% 触发达成
  8. 异步勋章判定
COMMIT
```

### 4.2 宠物互动 + 升级 → pending_draw = true

```text
BEGIN TRANSACTION
  1. SELECT item → cost_energy, gain_xp
  2. UPDATE users SET energy_balance -= :cost WHERE energy_balance >= :cost
  3. INSERT energy_logs
  4. SELECT pet; newXp = currentXp + gainXp; leveledUp = false
  5. IF newXp >= maxXp:
       level += 1
       newXp = newXp % maxXp  // 保留余数
       pendingDraw = true     // 布尔值，不累加
       leveledUp = true
  6. UPDATE pets SET currentXp=newXp, level=newLevel, pendingDraw=:pendingDraw
  7. INSERT interaction_logs
COMMIT
返回 { leveledUp, newLevel, newXp, pendingDraw }
// 前端收到 leveledUp=true → 升级动效 + 🎁亮起
```

### 4.3 愿望抽奖（消耗 pending_draw）

```text
BEGIN TRANSACTION
  1. SELECT pet WHERE userId=:userId AND pending_draw=true
     → 若 pending_draw=false → 抛"无可用抽奖机会"
  2. SELECT * FROM wishes WHERE userId=:userId AND status=0 AND deletedAt IS NULL
     → 若空 → 抛"奖池为空"
  3. 加权随机：totalWeight=Σ(weight), R=random(0,totalWeight), 遍历命中
  4. UPDATE wishes SET status=1, drawnAt=NOW() WHERE id=:hitId
  5. UPDATE pets SET pending_draw=false WHERE userId=:userId
  6. 异步勋章判定
COMMIT
返回 { drawnWish, pendingDraw: false, poolRemaining }
```

|----|------|
